       ((eq (car e) 'cdr)   (cdr    (eval. (cadr e) a)))

(defun is-prime (n)
  (cond ((= 2 n) t) 
        ((= 3 n) t) 
        ((evenp n) nil) 
        (t 
           (loop for i from 3 to (isqrt n) by 2
                 never (zerop (mod n i))))))
